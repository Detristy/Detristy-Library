# 1. 割点和割边
在学习双连通分量之前，我们学习有关割点和割边的内容：

割点：对于一个无向图，如果把一个点删除之后这个图的极大连通分量数增加了，那么这个点就是这个图的割点。

割边（桥）：对于一个无向图，如果删掉一条边后图的极大连通分量数增加了，那么这条边就是这张图的割边。
# 2. 双连通
在一张连通的无向图中，对于两个点 u 和 v ，如果无论删去哪一条边都不能使她们不连通，就说 u 和 v 边双连通。
在一张联通的无向图中，对于两个点 u 和 v ，如果无论删去除了 u 或 v 的哪个点都不能使它们不连通，则说 u 和 v 点双连通。

对于一个无向图中的极大边双连通的子图，我们称这个子图为一个 边双连通分量 e-DCC；
对于一个无向图中的极大点双连通的子图，我们称这个子图为一个点双连通分量 v-DCC。

边双连通具有传递性，自反性和对称性，因此边双连通分量是一个等价类；而点双连通不具有传递性。
# 3. Tarjan求边双连通分量
对于一张 $n$ 个结点，$m$ 条无向边的图，我们考虑求其的每一个边双连通分量并求边双连通分量的个数。
使用 Tarjan 算法来求边双连通分量，带着点 $u$ 和边 $e$ 一起跑 Tarjan。
我们记录每一个点的 $dfn$ 和 $low$ ，将每一个点入栈，我们遍历每一个点 $v$ 及其对应的边 $i$：

1. 如果 $dfn$ 为零，那么我们递归跑 Tarjan ，这里和求强连通分量是一样的；
2. 否则，如果 `i != e` 说明发现了一条重边，只要不是回边，它就有潜力拉低 low 值。

如果 $dfn=low$ ，说明我们要进行缩点操作，此时 $u$ 是这个边双连通分量的根，同时 $u$ 和其父节点的边 $e$ 是一个桥。

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

struct Edge {
    int to,id;
};

constexpr int N = 5e5+10;
int n,m,a,timer,top,ebcc_cnt,cnt_in;
vector<Edge> g[N];
vector<int> nodes[N];
int dfn[N],low[N],stk[N],ebcc[N],ebcc_siz[N],in_degree[N];
bool vis[N];

void tarjan(int u,int from_id) {
    dfn[u] = low[u] = ++timer;
    stk[++top] = u;
    for (auto [v,id] : g[u]) {
        if (!dfn[v]) {
            tarjan(v,id);
            low[u] = min(low[u],low[v]);
        }
        else if (id != from_id) low[u] = min(low[u],dfn[v]);
    }
    if (dfn[u] == low[u]) {
        ebcc_cnt++;
        while (true) {
            int node = stk[top--];
            ebcc[node] = ebcc_cnt;
            nodes[ebcc_cnt].push_back(node);
            if (node == u) break;
        }
    }
}

void solve() {
    cin >> n >> m;
    for (int i = 1 ; i <= m ; i++) {
        int u,v;
        cin >> u >> v;
        g[u].push_back({v,i});
        g[v].push_back({u,i});
    }
    for (int i = 1 ; i <= n ; i++) if (!dfn[i]) tarjan(i,-1);
    cout << ebcc_cnt << '\n';
    for (int i = 1 ; i <= ebcc_cnt ; i++) {
        cout << nodes[i].size() << ' ';
        for (int u : nodes[i]) cout << u << ' ';
        cout << '\n';
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int T = 1;
    //cin >> T:
    while (T--) solve();
    return 0;
}
```

# 4. Tarjan求割点
对于一张 $n$ 个结点，$m$ 条无向边的图，我们考虑求其割点。
遍历一颗 DFS 树，对于每一个遍历的点 $u$ 我们要对其进行检查：

1. 如果 $u$ 是根，并且根节点的孩子数量超过 1 个，那么如果将根节点删去，这些孩子就无法连通，此时 $u$ 是割点
2. 如果 $u$ 不是根，并且其子节点 $v$ 满足 `low[v] >= dfn[u]` ，那么说明 $v$ 无法回溯到 $u$ 更早的祖先，此时 $u$ 就连接着 $v$ 和外界其他店，删去 $u$ 会导致 $v$ 和图的其他部分不连通，此时 $u$ 是割点

我们可以跑 tarjan 来实现求割点的过程：

```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long

constexpr int N = 2e4+10;
vector<int> g[N];
int dfn[N],low[N],timer;
bool cut[N];
int n,m;

void tarjan(int u,int fa,int r) {
    dfn[u] = low[u] = ++timer;
    int child = 0; //记录DFS树中每个点的子节点个数
    for (int v : g[u]) {
        if (!dfn[v]) {
            child++;
            tarjan(v,u,r);
            low[u] = min(low[u],low[v]);
            if (u != r && dfn[u] <= low[v]) cut[u] = true; //判断非根节点是否是割点
        }
        else if (v != fa) low[u] = min(low[u],dfn[v]);
    }
    if (u == r && child > 1) cut[u] = true; //判断根节点是否是割点
}

void solve() {
    cin >> n >> m;
    for (int i = 1 ; i <= m ; i++) {
        int u,v; cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    for (int i = 1 ; i <= n ; i++) if (!dfn[i]) tarjan(i,0,i);
    int cnt = 0;
    for (int i = 1 ; i <= n ; i++) if (cut[i]) cnt++;
    cout << cnt << '\n';
    for (int i = 1 ; i <= n ; i++) if (cut[i]) cout << i << ' ';
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    //cin >> T;
    while (T--) solve();
    return 0;
}
```

# 5. Tarjan求点双连通分量
对于一张 $n$ 个结点，$m$ 条无向边的图，我们考虑求其的每一个点双连通分量并求点双连通分量的个数。
我们在 DFS 树中讨论点双连通分量的判定，如果我们发现一个点 $u$ 和它的子节点 $v$ 满足

$$ low_v \geq dfn_u $$

时，意味着 $v$ 以及 $v$ 的子节点最高只能回溯到 $u$ 而无法回到 $u$ 的更早的祖先，此时 $u$ 就是一个割点，此时 $v$ 为根节点的子树和 $u$ 本身共同构成了一个点双连通分量。
需要注意的是，点 $u$ 作为割点，此时被两个点双连通分量所共用，因此我们在跑tarjan的时候就不出栈了，用于判断另一个点双连通分量。

这里还有两个坑点：

1. 一个单独的孤立点就是一个点双连通分量；
2. 当一个点只有自环的时候，视作一个孤立点，也就是一个点双连通分量。

第二个坑点可以通过在输入的时候直接归类到第一类（如果边的两端相等，即自环，就不建边，此时就相当于是孤立点）。

```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long

constexpr int N = 5e5+10;
int n,m;
vector<int> g[N],vbcc[N];
int dfn[N],low[N],stk[N],top,timer,vbcc_cnt;

void tarjan(int u,int fa,int r) {
    dfn[u] = low[u] = ++timer;
    stk[++top] = u;
    //判孤立点
    if (u == r && g[u].empty()) {
        vbcc[++vbcc_cnt].push_back(u);
        return;
    }
    for (int v : g[u]) {
        if (v == fa) continue;
        if (!dfn[v]) {
            tarjan(v,u,r);
            low[u] = min(low[u],low[v]);
            if (low[v] >= dfn[u]) {
                vbcc_cnt++;
                while (true) {
                    int node = stk[top--];
                    vbcc[vbcc_cnt].push_back(node);
                    if (node == v) break;
                }
                vbcc[vbcc_cnt].push_back(u);
            }
        }
        else {
            low[u] = min(low[u],dfn[v]);
        }
    }
}

void solve() {
    cin >> n >> m;
    for (int i = 1 ; i <= m ; i++) {
        int u,v;
        cin >> u >> v;
        if (u == v) continue; //判自环
        g[u].push_back(v);
        g[v].push_back(u);
    }
    for (int i = 1 ; i <= n ; i++) {
        if (!dfn[i]) tarjan(i,i,i);
    }
    cout << vbcc_cnt << '\n';
    for (int i = 1 ; i <= vbcc_cnt ; i++) {
        cout << vbcc[i].size() << ' ';
        for (auto u : vbcc[i]) {
            cout << u << ' ';
        }
        cout << '\n';
    }
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    //cin >> T;
    while (T--) solve();
    return 0;
}
```
