# 1. 差分约束

差分约束是一种特殊的 n 元一次不等式组，形如：
$$  
\begin{cases} x_{c_1}-x_{c'_1}\leq y_1 \\x_{c_2}-x_{c'_2} \leq y_2 \\ \cdots\\ x_{c_m} - x_{c'_m}\leq y_m\end{cases}
$$

我们的任务是求出一组解，满足上面所有的不等式，否则我们需要判断其无解，不难发现，我们能将上式转换成：

$$  
\begin{cases} x_{c_1}\leq y_1 +x_{c'_1}\\x_{c_2} \leq y_2 +x_{c'_2}\\ \cdots\\ x_{c_m} \leq y_m + x_{c'_m}\end{cases}  
$$


这和最短路中的三角不等式 $dis[u]\leq dis[u]+w$ 相似，因此我们可以把每个变量 $x_{i}$ 看作图中的一个节点，对于每一个不等式 $x_{i}-x_{j}\leq c_{k}$，可以视作从 $j$ 到 $i$ 的一条长度为 $c_{k}$ 的有向边，建图即可。如果出现负环就说明无解。

```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long

constexpr int N = 5e3+10;
vector<pair<int,int>> g[N];
int n,m,dis[N],cnt[N];
bool vis[N];
queue<int> q;

bool spfa(int t,int x) {
    for (int i = 0 ; i <= t ; i++) {
        dis[i] = 0x3f3f3f3f;
        vis[i] = false;
        cnt[i] = 0;
    }
    q.push(x);
    vis[x] = true;
    dis[x] = 0;
    cnt[x] = 1;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        vis[u] = false;
        for (auto [v,w] : g[u]) {
            if (dis[v] <= dis[u] + w) continue;
            dis[v] = dis[u] + w;
            if (!vis[v]) {
                vis[v] = true;
                cnt[v]++;
                if (cnt[v] > t) return false;
                q.push(v);
            }
        }
    }
    return true;
}

void solve() {
    cin >> n >> m;
    for (int i = 1; i <= m ; i++) {
        int u,v,w;
        cin >> u >> v >> w;
        g[v].push_back({u,w});
    }
    for (int i = 1 ;i <= n ; i++) g[0].push_back({i,0});
    if (!spfa(n+1,0)) cout << "NO\n";
    else {
        for (int i = 1 ; i <= n ; i++) cout << dis[i] << ' ';
        cout << '\n';
    }
}

signed main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    //cin >> T;
    while (T--) solve();
    return 0;
}
```
