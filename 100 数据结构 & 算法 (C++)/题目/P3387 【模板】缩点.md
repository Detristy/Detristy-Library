# 2. 分析
对于这题我们被要求求一条点权之和最大的路径，对于一个DAG而言，我们可以直接使用拓扑排序+DP来做，但是题目给我们的是一个有向有环图，而题目允许我们多次经过一条边或者一个点，因此我们可以将一个环直接视作一个点。

这里涉及到了缩点的操作，实现的前提是在这张图中找到所有的强连通分量，对于这部分的内容我们可以通过Tarjan算法来实现。

我们用数组模拟一个栈，接着开始从根节点开始跑Tarjan。

接下来我们考虑如何处理每一个强连通分量，我们现在知道每一个强连通分量本质上都将被缩成一个点，那么我们不妨给这个强联通分量中的每一个点都赋一个相同的scc编号值，这样当我们遇到两个scc编号相同的点我们就能知道这两个点都属于同一个强连通分量中。

我们用一个timer来记录每个点被搜索的顺序，故而我们每搜索一个点，timer都+1 。

时间复杂度 $O(n+m)$ 。

我们通过 Tarjan 将每个强连通分量缩成点，处理结束之后就可以用这些点建图了，因为我们已经将所有的强连通分量都已经变成单独的点了，所以我们此时就能保证此时的图是DAG，于是要求最大点权值的路径，我们就使用拓扑排序+动态规划的方式来解决，就是在跑拓扑排序的同时跑DP

定义 $dp[v]$ 为以 $v$ 结尾的路径的最大点权和，我们从 $u\to v$ 的状态转移方程写为：

$$ dp[v]=\max(dp[v],dp[u]+p[v]) $$

其中 $p[v]$ 是点 v 的权值

# 3. 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int n,m,timer;
constexpr int N  = 1e4+10;
vector<int> g[N],dag[N];
int w[N],dfn[N],low[N];
int stk[N],top,scc[N],scc_cnt,scc_w[N];
bool vis[N];
int dist[N],in_degree[N];

void tarjan(int u) {
    dfn[u] = low[u] = ++timer;
    stk[++top] = u;
    vis[u] = true;
    for (int v : g[u]) {
        if (!dfn[v]) {
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else if (vis[v]) low[u] = min(low[u],dfn[v]);
    }
    if (dfn[u] == low[u]) {
        scc_cnt++;
        while (true) {
            int v = stk[top--];
            vis[v] = false;
            scc[v] = scc_cnt;
            scc_w[scc_cnt] += w[v];
            if (u == v) break;
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1 ; i <= n ; i++) cin >> w[i];
    for (int i = 1 ; i <= m ; i++) {
        int u,v; cin >> u >> v;
        g[u].push_back(v);
    }
    for (int i = 1 ; i <= n ; i++) if (!dfn[i]) tarjan(i);
    for (int i = 1 ; i <= n ; i++) {
        for (int v : g[i]) {
            if (scc[i] != scc[v]) {
                dag[scc[i]].push_back(scc[v]);
                in_degree[scc[v]]++;
            }
        }
    }
    queue<int> q;
    for (int i = 1 ; i <= scc_cnt ; i++) {
        dist[i] = scc_w[i];
        if (in_degree[i] == 0) q.push(i);
    }
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : dag[u]) {
            dist[v] = max(dist[v],dist[u] + scc_w[v]);
            if (--in_degree[v] == 0) q.push(v);
        }
    }
    int ans = 0;
    for (int i = 1 ; i <= scc_cnt ; i++) ans = max(ans,dist[i]);
    cout << ans << '\n';
}
```
