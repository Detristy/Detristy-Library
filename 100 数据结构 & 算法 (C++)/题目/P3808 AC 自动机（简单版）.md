#include<bits/stdc++.h>
using namespace std;
#define int long long

const int N = 1e6+10;
int n;

int get_id(char c) {
    if (c >= 'A' && c <= 'Z') return c - 'A';
    if (c >= 'a' && c <= 'z') return c - 'a' + 26;
    return c - '0' + 52;
}

struct ACAuto {
    int nxt[N][62];
    int cnt[N];
    int fail[N];
    bool vis[N];
    int idx;

    void clear() {
        for (int i = 0; i <= idx; i++) {
            for (int j = 0; j < 62; j++) nxt[i][j] = 0;
            cnt[i] = 0;
            fail[i] = 0;
            vis[i] = false;
        }
        idx = 0;
    }

    void insert(string s) {
        int u = 0;
        for (char c : s) {
            int v = get_id(c);
            if (!nxt[u][v]) nxt[u][v] = ++idx;
            u = nxt[u][v];
        }
        cnt[u]++;
    }

    void findFail() {
        queue<int> q;
        for (int i = 0 ; i < 62 ; i++) {
            if (nxt[0][i]) {
                fail[nxt[0][i]] = 0;
                q.push(nxt[0][i]);
            }
        }
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int i = 0 ; i < 62 ; i++) {
                if (nxt[u][i]) {
                    fail[nxt[u][i]] = nxt[fail[u]][i];
                    q.push(nxt[u][i]);
                }
                else {
                    nxt[u][i] = nxt[fail[u]][i];
                }
            }
        }
    }

    int match(string s) {
        int u = 0,ans = 0;
        for (char c : s) {
            int v = get_id(c);
            u = nxt[u][v];
            int tmp = u;
            while (tmp && !vis[tmp]) {
                ans += cnt[tmp];
                vis[tmp] = true;
                tmp = fail[tmp];
            }
        }
        return ans;
    }
} t;

void solve() {
    cin >> n;
    t.clear();
    for (int i = 1 ; i <= n ; i++) {
        string lib;
        cin >> lib;
        t.insert(lib);
    }
    t.findFail();
    string s; cin >> s;
    int ans = t.match(s);
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int T = 1;
    //cin >> T;
    while (T--) solve();
    return 0;
}
