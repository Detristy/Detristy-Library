# 代码

```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long

constexpr int N = 1e6+10;
string s,lib;
int slen,liblen;
int nxt[N];

void kmp() {
    int t = 0;
    for (int i = 1 ; i < liblen ; i++) {
        while (t && lib[t] != lib[i]) t = nxt[t-1];
        if (lib[t] == lib[i]) t++;
        nxt[i] = t;
    }
}

void search() {
    int t = 0;
    for(int i = 0 ; i < slen ; i++){
        while(t && s[i] != lib[t]) t = nxt[t-1];
        if(s[i] == lib[t]) t++;
        if(t == liblen) {
            cout << i-liblen+2 << endl;
            t = nxt[t-1];
        }
    }
}

void solve() {
    cin >> s >> lib;
    liblen = lib.length();
    slen = s.length();
    kmp();
    search();
    for(int i = 0; i < liblen; i++) {
        cout << nxt[i] << ' ';
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int T = 1;
    //cin >> T;
    while (T--) solve();
    return 0;
}
```
