# 2. 题解

题目要求我们维护两个区间修改和一个区间查询。
一个区间修改是区间加，一个区间修改是区间乘。

也就是说，不同节点间的关系呈一个一次函数 $f(x)=kx + b$ 的仿射变换关系。

这题包含两种操作，明显我们要为之分别维护一个标记：我们设 $lm$ 为乘法， $la$ 为加法，那么对于一个节点 $k$ 的值 $V_k$ ，我们有：

$$
V_{current}=V_{original}\times lm +len\times \tag{1.1}
$$

其中 $len$ 是当前区间的长度。

对于这个运算，我们显然知道乘法是优先于加法的。
接下来我们先从简单的开始：执行操作 2 。

根据我们推导的公式 $(1.1)$ 可以得到，我们只需要更新加法的部分即可，假设我们对区间每个数加上 $K$ ，于是有：

$$
V_{current}=V_{original}\times lm +len\times (la+K) \tag{1.2}
$$

我们发现这个时候只有 $la$ 发生了变化，而 $lm$ 没有。然后我们来看看操作 1 是怎么运作的。

我们假设对每个区间每个数乘上 $K$ ，那么有：

$$
V_{current}=(V_{original}\times lm +len\times la)\times K \tag{1.3}
$$

我们用分配律拆开，得到：

$$
V_{current}=(V_{original}\times lm )\times K+(len\times la)\times K \tag{1.4}
$$

也就是说当我们执行乘法操作的时候，我们维护的标记 $la$ 和 $lm$ 都发生了变化（乘上了 $K$ ）。

我们需要重新写 pushdown 的逻辑：
```cpp
    void pushdown(int x,int l,int r) {
        if (lm[x] == 1 && !la[x]) return;
        int mid = (l+r)/2;
        //函数关系
        rt[x*2] = lm[x]*rt[x*2]+(mid-l+1)*la[x]; 
        rt[x*2+1] = lm[x]*rt[x*2+1]+(r-mid)*la[x];
        rt[x*2] %= m;
        rt[x*2+1] %= m;
				
				//对lm进行更新，当 lm[x]=1 的时候两边相等
        lm[x*2] *= lm[x];
        lm[x*2+1] *= lm[x];
        lm[x*2] %= m;
        lm[x*2+1] %= m;
				
				//对la进行更新，la既进行 +运算，也进行 *运算
        la[x*2] *= lm[x];
        la[x*2+1] *= lm[x];
        la[x*2] += la[x];
        la[x*2+1] += la[x];
        la[x*2] %= m;
        la[x*2+1] %= m;

        lm[x] = 1;
        la[x] = 0;
    }
```

注意，因为题目的答案需要对 $m$ 取模，所以这里每一次运算都对 $m$ 取一次模。

以及对于乘法运算，其单位元是 1，所以我们要将 $lm$ 的标记初始化为 1 。

让我们考虑这样的嵌套，对于上一个节点，其对应的函数关系是： $f(x)=k'x+b'$ ，而对于目前的节点，其对应的函数关系是 $g(x)=kx_1+b$ 。

其中 $x_1$ 的值就是上个节点的值，也就是 $x_1=f(x)$，我们代入就有： $g(x)=kf(x)+b$ ，我们展开：

$$
f(x)=kk'x+kb'+b
$$

在pushdown函数中，`lm[p]*=k` 就对应着 $kk'$ ，而 `la[p]*=k , la[p]+=k` 就对应着 $kb'+b$

接着我们就可以愉快的补完剩下的代码了，记得在每次运算的时候对 m 取模。

# 3. 代码

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int n,q,m;
const int N = 1e5+10;
ll a[N];

struct segment_tree {
    ll rt[N*4];
    ll lm[N*4],la[N*4];

    void pushdown(int x,int l,int r) {
        if (lm[x] == 1 && !la[x]) return;
        int mid = (l+r)/2;
        rt[x*2] = lm[x]*rt[x*2]+(mid-l+1)*la[x];
        rt[x*2+1] = lm[x]*rt[x*2+1]+(r-mid)*la[x];
        rt[x*2] %= m;
        rt[x*2+1] %= m;

        lm[x*2] *= lm[x];
        lm[x*2+1] *= lm[x];
        lm[x*2] %= m;
        lm[x*2+1] %= m;

        la[x*2] *= lm[x];
        la[x*2+1] *= lm[x];
        la[x*2] += la[x];
        la[x*2+1] += la[x];
        la[x*2] %= m;
        la[x*2+1] %= m;

        lm[x] = 1;
        la[x] = 0;
    }
    void build(int x = 1,int l = 1,int r = n) {
        lm[x]=1;
        if (l == r) {
            rt[x] = a[l];
            return;
        }
        ll mid = (l+r)/2;
        build(x*2,l,mid);
        build(x*2+1,mid+1,r);
        rt[x] = rt[x*2] + rt[x*2+1];
        rt[x] %= m;
    }
    void mergeS_Add(int ql,int qr,ll val,int x = 1,int l = 1,int r = n) {
        if (ql <= l && r <= qr) {
            rt[x] += (r-l+1) * val, rt[x] %= m;
            la[x] += val, la[x] %= m;
            return;
        }
        pushdown(x,l,r);
        ll mid = (l+r)/2;
        if (ql <= mid) mergeS_Add(ql,qr,val,x*2,l,mid);
        if (mid < qr) mergeS_Add(ql,qr,val,x*2+1,mid+1,r);
        rt[x] = rt[x*2] + rt[x*2+1];
        rt[x] %= m;
    }
    void mergeS_Mul(int ql,int qr,ll val,int x = 1 ,int l = 1,int r = n) {
        if (ql <= l && r <= qr) {
            rt[x] *= val, rt[x] %= m;
            la[x] *= val, la[x] %= m;
            lm[x] *= val, lm[x] %= m;
            return;
        }
        pushdown(x,l,r);
        ll mid = (l+r)/2;
        if (ql <= mid) mergeS_Mul(ql,qr,val,x*2,l,mid);
        if (mid < qr) mergeS_Mul(ql,qr,val,x*2+1,mid+1,r);
        rt[x] = rt[x*2] + rt[x*2+1];
        rt[x] %= m;
    }
    ll query(int ql,int qr,int x = 1,int l = 1,int r = n) {
        if (ql <= l && r <= qr) return rt[x];
        pushdown(x,l,r);
        ll mid = (l+r)/2;
        ll ans = 0;
        if (ql <= mid) ans += query(ql,qr,x*2,l,mid);
        if (mid < qr) ans += query(ql,qr,x*2+1,mid+1,r);
        return ans%m;
    }
}t;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    cin >> n >> q >> m;
    for (int i = 1 ; i <= n ; i++) cin >> a[i];
    t.build();
    while (q--) {
        ll op,x,y,k;
        cin >> op >> x >> y;
        if (op == 1) {
            cin >> k;
            t.mergeS_Mul(x,y,k);
        }
        else if (op == 2) {
            cin >> k;
            t.mergeS_Add(x,y,k);
        }
        else if (op == 3) {
            cout << t.query(x,y) << "\n";
        }
    }
}
```
