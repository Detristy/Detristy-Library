# 分析

这题要求 $S_{1}\sim S_{N}$ 中有多少个字符串是 $T$ 的前缀，所以我们要对 [[104-2 trie]] 板子进行处理：

1. 在跑 insert 的时候，我们要对末尾的字符打一个标记，定义 $end[x]$ 为字符串的最后一个字符在结点 $x$ 处。
2. 在跑字符串 $T$ 的 query 的时候，我们要加上以当前字符为结尾的前缀个数，如果为 $0$ 的话也不会有影响。

$end[x]$ 是一个 Trie 上常用的标记写法。 

# 代码
```cpp
#include<bits/stdc++.h>  
using namespace std;  
  
const int N = 1e6+5;  
  
int get_id(char c) {  
    if (c >= 'A' && c <= 'Z') return c - 'A';  
    if (c >= 'a' && c <= 'z') return c - 'a' + 26;  
    return c - '0' + 52;  
}  
  
struct Trie {  
    int nxt[N][62];  
    int end[N];  
    int cnt[N];  
    int idx;  
  
    void clear() {  
        for (int i = 0; i <= idx; i++) {  
            for (int j = 0; j < 62; j++) nxt[i][j] = 0;  
            cnt[i] = 0;  
            end[i] = 0;  
        }  
        idx = 0;  
    }  
  
    void insert(string s) {  
        int u = 0;  
        for (int i = 0 ; i < s.length(); i++) {  
            char c = s[i];  
            int v = get_id(c);  
            if (!nxt[u][v]) nxt[u][v] = ++idx;  
            u = nxt[u][v];  
            cnt[u]++;  
        }  
        end[u]++;  
    }  
  
    int query(string s) {  
        int u = 0;  
        int res = 0;  
        for (char c : s) {  
            int v = get_id(c);  
            if (!nxt[u][v]) return res;  
            u = nxt[u][v];  
            res += end[u];  
        }  
        return res;  
    }  
} t;  
  
void solve() {  
    int n, q;  
    cin >> n >> q;  
    t.clear();  
    for (int i = 0; i < n; i++) {  
        string s; cin >> s;  
        t.insert(s);  
    }  
    while (q--) {  
        string s; cin >> s;  
        cout << t.query(s) << "\n";  
    }  
}  
  
int main() {  
    ios::sync_with_stdio(false);  
    cin.tie(NULL);  
  
    int T = 1;  
    //cin >> T;  
    while (T--) solve();  
    return 0;  
}
```
