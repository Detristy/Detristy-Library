# 代码

```cpp
#include <bits/stdc++.h>  
using namespace std;  
#define int long long  
#define detristy ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);  
  
map<string,string> fa;  
int n,m,k;  
  
string find(string s) {  
    if (fa.find(s) == fa.end()) {  
        fa[s] = s;  
        return s;  
    }  
    return fa[s] == s ? s : fa[s] = find(fa[s]);  
}  
  
void solve() {  
    cin >> n >> m;  
    for (int i = 1 ; i <= n ; i++) {  
        string s; cin >> s;  
    }  
    for (int i = 1 ; i <= m ; i++) {  
        string s1,s2; cin >> s1 >> s2;  
        if (find(s1) != find(s2)) {  
            fa[find(s1)] = find(s2);  
        }  
    }  
    cin >> k;  
    for (int i = 1 ; i <= k ; i++) {  
        string s1,s2; cin >> s1 >> s2;  
        if (find(s1) != find(s2)) cout << "No.\n";  
        else cout << "Yes.\n";  
    }  
}  
  
signed main() {  
    detristy  
    int T = 1;  
    //cin >> T;  
    while (T--) solve();  
    return 0;  
}
```
